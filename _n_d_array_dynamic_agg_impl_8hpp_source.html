36 DType result{};
37 auto size = m_ArrayImpl.size();
38 for (std::size_t i = 0; i < size; ++i) {
39 const auto& element = get(i);
40 result += element;
41 }

77 // Cumulative sum of the elements
78 template<typename DType, typename Storage>
79 inline NDArrayDynamic<DType, Storage> NDArrayDynamic<DType, Storage>::cumsum() const {
80 Shape sh = shape();
81 sh.flatten();
82 NDArrayDynamic<DType, Storage> result{sh};
83 DType sum = 0;
84 auto size = m_ArrayImpl.size();
85 for (std::size_t i = 0; i < size; ++i) {
86 const auto& element = get(i);
87 sum += element;
88 result.m_ArrayImpl.set(i, sum);
89 }
90 return result;

105 // Median
106 template<typename DType, typename Storage>
107 inline DType NDArrayDynamic<DType, Storage>::median() const {
108 auto size = m_ArrayImpl.size();
109 if (size == 0)
110 return 0;
111 auto v = m_ArrayImpl.m_Impl;
112 if (size % 2 == 0) {
113 std::nth_element(v.begin(),
114 v.begin() + size / 2,
115 v.end());
116
117 std::nth_element(v.begin(),
118 v.begin() + (size - 1) / 2,
119 v.end());
120
121 // Find the average of values at indices size / 2 and (size - 1) / 2
122 return static_cast<DType>(v[(size - 1) / 2] + v[(size / 2)] / 2.0);
123 }
124 std::nth_element(v.begin(),
125 v.begin() + size / 2,
126 v.end());
127 return static_cast<DType>(v[size / 2]);
128 }

130 template<typename DType, typename Storage>
131 inline static DType vectorCov(const NDArrayDynamic<DType, Storage>& v1, const NDArrayDynamic<DType, Storage>& v2) {
132 auto sh1 = v1.shape();
133 if (sh1.size() != 1)
134 throw std::runtime_error("Only 1D arrays supported");
135
136 auto sh2 = v2.shape();
137 if (sh2.size() != 1)
138 throw std::runtime_error("Only 1D arrays supported");
139
140 if (v1.len() != v2.len()) {
141 throw std::runtime_error("Sizes are different");
142 }
143
144 auto v1_mean = v1.mean();
145 auto v2_mean = v2.mean();
146
147 DType sum = 0;
148
149 for (Size i = 0; i < v1.len(); ++i) {
150 sum += ((v1[i] - v1_mean) * (v2[i] - v2_mean));
151 }
152
153 return sum / (v1.len() - 1);
154 }

156 template<typename DType, typename Storage>
157 inline NDArrayDynamic<DType, Storage> NDArrayDynamic<DType, Storage>::cov() const {
158 auto sh = shape();
159 if (sh.size() == 0)
160 return 0;
161 if (sh.size() != 1 && sh.size() != 2)
162 throw std::runtime_error("Only 1D and 2D arrays supported");
163
164 if (sh.size() == 1)
165 return NDArrayDynamic<DType, Storage>{1.0};
166
167 Shape resultShape({len(), len()});
168 NDArrayDynamic<DType, Storage> result{resultShape};
169 for (Size i = 0; i < len(); ++i) {
170 Shape subResultShape({1, len()});
171 NDArrayDynamic<DType, Storage> subResult{subResultShape};
172 for (Size j = 0; j < len(); ++j) {
173 auto subArray1 = at(i);
174 auto subArray2 = at(j);
175 DType c = vectorCov(subArray1, subArray2);
176 set(subResult, j, c);
177 }
178 set(result, i, subResult);
179 }
180 return result;
181 }

183 template<typename DType, typename Storage>
184 inline static DType vectorCorr(const NDArrayDynamic<DType, Storage>& v1, const NDArrayDynamic<DType, Storage>& v2) {
185 auto sh1 = v1.shape();
186 if (sh1.size() != 1)
187 throw std::runtime_error("Only 1D arrays supported");
188
189 auto sh2 = v2.shape();
190 if (sh2.size() != 1)
191 throw std::runtime_error("Only 1D arrays supported");
192
193 if (v1.len() != v2.len()) {
194 throw std::runtime_error("Sizes are different");
195 }
196
197 DType sum1 = 0;
198 DType sum2 = 0;
199 DType sum12 = 0;
200 DType squareSum1 = 0;
201 DType squareSum2 = 0;
202
203 Size n = v1.len();
204
205 for (Size i = 0; i < n; i++) {
206 sum1 += v1.get(i);
207 sum2 += v2.get(i);
208 sum12 += sum12 + v1.get(i) * v2.get(i);
209
210 squareSum1 += v1.get(i) * v1.get(i);
211 squareSum2 += v2.get(i) * v2.get(i);
212 }
213
214 DType corr = static_cast<DType>(n * sum12 - sum1 * sum2)
215 / sqrt((n * squareSum1 - sum1 * sum1)
216 * (n * squareSum2 - sum2 * sum2));
217
218 return corr;
219 }

221 // Correlation coefficient
222 template<typename DType, typename Storage>
223 inline NDArrayDynamic<DType, Storage> NDArrayDynamic<DType, Storage>::corrcoef() const {
224 auto sh = shape();
225 if (sh.size() == 0)
226 return NDArrayDynamic<DType, Storage>{};
227
228 if (sh.size() != 1 && sh.size() != 2)
229 throw std::runtime_error("Only 1D and 2D arrays supported");
230
231 if (sh.size() == 1)
232 return NDArrayDynamic<DType, Storage>{1};
233
234 Shape resultShape({len(), len()});
235 NDArrayDynamic<DType, Storage> result{resultShape};
236 for (Size i = 0; i < len(); ++i) {
237 Shape subResultShape({1, len()});
238 NDArrayDynamic<DType, Storage> subResult{subResultShape};
239 for (Size j = 0; j < len(); ++j) {
240 auto subArray1 = at(i);
241 auto subArray2 = at(j);
242 DType c = vectorCorr(subArray1, subArray2);
243 subResult.set(j, c);
244 }
245 result.set(i, subResult);
246 }
247 return result;
248 }

250 // Standard deviation
251 template<typename DType, typename Storage>
252 inline DType NDArrayDynamic<DType, Storage>::std_() const {
253 NDArrayDynamic<DType, Storage> x{shape()};
254 DType m = mean();
255 for (std::size_t i = 0; i < m_ArrayImpl.size(); ++i) {
256 auto a = abs(get(i) - m);
257 x.set(i, a * a);
258 }
259 return std::sqrt(x.mean());
260 }
