33 // Returns a new Array with axes transposed.
34 template<typename DType, typename Storage>
35 NDArrayDynamic<DType, Storage> NDArrayDynamic<DType, Storage>::transpose() const {
36 auto sh = shape();
37 sh.transpose();
38 NDArrayDynamic<DType, Storage> result{sh};
39
40 Size offset = 0;
41 std::size_t column = 0;
42 for (Size i = 0; i < size(); ++i) {
43 result.m_ArrayImpl.m_Impl[i] = m_ArrayImpl.m_Impl[offset + column];
44 offset += sh[0];
45 if (offset >= size()) {
46 offset = 0;
47 column++;
48 }
49 }
50
51 return result;
52 }

81 // Return a new array with new size
82 template<typename DType, typename Storage>
83 NDArrayDynamic<DType, Storage> NDArrayDynamic<DType, Storage>::resize(const Shape& shape) const {
84 Size newSize = ndarray::internal::calcSizeByShape(shape);
85 Size copySize = std::min(size(), newSize);
86 NDArrayDynamic<DType, Storage> result{shape};
87 for (Size offset = 0; offset < newSize; offset += copySize) {
88 std::copy(m_ArrayImpl.m_Impl.begin(), m_ArrayImpl.m_Impl.begin() + copySize, result.m_ArrayImpl.m_Impl.begin() + offset);
89 }
90 return result;

93 // Append items to an array
94 template<typename DType, typename Storage>
95 NDArrayDynamic<DType, Storage> NDArrayDynamic<DType, Storage>::append(const NDArrayDynamic<DType, Storage>& array) const {
96 if (array.size() == 0)
97 return *this;
98 auto size1 = ndarray::internal::calcSizeByShape(shape());
99 auto size2 = ndarray::internal::calcSizeByShape(array.shape());
100 Shape sh{size1 + size2};
101 NDArrayDynamic<DType, Storage> result(sh);
102 std::copy(m_ArrayImpl.m_Impl.begin(), m_ArrayImpl.m_Impl.end(), result.m_ArrayImpl.m_Impl.begin());
103 std::copy(array.m_ArrayImpl.m_Impl.begin(), array.m_ArrayImpl.m_Impl.end(), result.m_ArrayImpl.m_Impl.begin() + m_ArrayImpl.m_Impl.size());
104 return result;

108 template<typename DType, typename Storage>
109 NDArrayDynamic<DType, Storage> NDArrayDynamic<DType, Storage>::insert(Size index, const NDArrayDynamic<DType, Storage>& array) const {
110 auto size1 = size();
111 if (index > size1) {
112 throw std::runtime_error("Index exceeds array bounds");
113 }
114 if (array.size() == 0)
115 return *this;
116 auto size2 = array.size();

118 NDArrayDynamic<DType, Storage> result(sh);
119 std::copy(m_ArrayImpl.m_Impl.begin(), m_ArrayImpl.m_Impl.begin() + index, result.m_ArrayImpl.m_Impl.begin());
120 std::copy(array.m_ArrayImpl.m_Impl.begin(), array.m_ArrayImpl.m_Impl.end(), result.m_ArrayImpl.m_Impl.begin() + index);
121 std::copy(m_ArrayImpl.m_Impl.begin() + index, m_ArrayImpl.m_Impl.end(), result.m_ArrayImpl.m_Impl.begin() + index + array.m_ArrayImpl.m_Impl.size());
122 return result;

126 template<typename DType, typename Storage>
127 NDArrayDynamic<DType, Storage> NDArrayDynamic<DType, Storage>::del(Size index) const {
128 auto sz = size();
129 if (sz == 0)
130 throw std::runtime_error("Cannot del from an empty array");
131 if (index > sz) {
132 throw std::runtime_error("Index exceeds array bounds");
133 }
134 Shape sh{sz - 1};
135 NDArrayDynamic<DType, Storage> result(sh);
136 std::copy(m_ArrayImpl.m_Impl.begin(), m_ArrayImpl.m_Impl.begin() + index, result.m_ArrayImpl.m_Impl.begin());
137 if (index < sz - 1) {
138 std::copy(m_ArrayImpl.m_Impl.begin() + index + 1, m_ArrayImpl.m_Impl.end(),
139 result.m_ArrayImpl.m_Impl.begin() + index);

158 // Stack arrays vertically (rowwise)
159 template<typename DType, typename Storage>
160 NDArrayDynamic<DType, Storage> NDArrayDynamic<DType, Storage>::vstack(const NDArrayDynamic<DType, Storage>& array) const {
161 if (size() == 0) {
162 return array;
163 }
164 if (array.size() == 0) {
165 return *this;
166 }
167 // Both are not empty
168 Shape sh1 = shape();
169 Shape sh2 = array.shape();
170 if (sh1.size() != sh2.size())
171 throw std::runtime_error("Number of dims should be equal");
172 // All the dims except the first should be equal
173 for (std::size_t i = 1; i < sh1.size(); ++i) {
174 if (sh1[i] != sh2[i])
175 throw std::runtime_error("All the dims except the first should be equal");
176 }
177 Shape sh = shape();
178 sh[0] = sh1[0] + sh2[0];
179 NDArrayDynamic<DType, Storage> result{array};
180 result.m_ArrayImpl.m_Shape = sh;
181 return result.insert(0, *this);
182 }

190 // Stack arrays horizontally (columnwise)
191 template<typename DType, typename Storage>
192 NDArrayDynamic<DType, Storage> NDArrayDynamic<DType, Storage>::hstack(const NDArrayDynamic<DType, Storage>& array) const {
193 if (size() == 0) {
194 return array;
195 }
196 if (array.size() == 0) {
197 return *this;
198 }
199 // Both are not empty
200 Shape sh1 = shape();
201 Shape sh2 = array.shape();
202 if (sh1.size() != sh2.size())
203 throw std::runtime_error("Number of dims should be equal");
204 if (sh1.size() == 1) {
205 //concatenation along 1st axis
206 return concatenate(array);
207 }
208 // All the dims except the first should be equal
209 for (std::size_t i = 0; i < sh1.size(); ++i) {
210 if (i != 1 && sh1[i] != sh2[i])
211 throw std::runtime_error("All the dims except the second should be equal");
212 }
213 Shape sh = shape();
214 sh[1] = sh1[1] + sh2[1];
215 NDArrayDynamic<DType, Storage> result{sh};
216 std::size_t destOffset = 0;
217 for (Size y = 0; y < sh[0]; ++y) {
218 std::copy(m_ArrayImpl.m_Impl.begin() + y * sh1[1],
219 m_ArrayImpl.m_Impl.begin() + (y + 1) * sh1[1],
220 result.m_ArrayImpl.m_Impl.begin() + destOffset);
221 destOffset += sh1[1];
222 std::copy(array.m_ArrayImpl.m_Impl.begin() + y * sh2[1],
223 array.m_ArrayImpl.m_Impl.begin() + (y + 1) * sh2[1],
224 result.m_ArrayImpl.m_Impl.begin() + destOffset);
225 destOffset += sh2[1];
226 }
227 return result;
228 }

230 // Create stacked columnwise arrays
231 template<typename DType, typename Storage>
232 NDArrayDynamic<DType, Storage> NDArrayDynamic<DType, Storage>::column_stack(const NDArrayDynamic<DType, Storage>& array) const {
233 Shape sh1 = shape();
234 Shape sh2 = array.shape();
235 if (sh1.size() != 1 || sh2.size() != 1)
236 throw std::runtime_error("Arrays must be 1D");
237 if (sh1[0] != sh2[0])
238 throw std::runtime_error("Arrays have different dimensions");
239 Size length = sh1[0];
240 Shape sh{length, 2};
241 NDArrayDynamic<DType, Storage> result{sh};
242 std::size_t index1 = 0;
243 std::size_t index2 = 0;
244 Size i = 0;
245 while (i < size() + array.size()) {
246 result.set(i++, get(index1++));
247 result.set(i++, array.get(index2++));
248 }
249 return result;
250 }

259 // Split the array horizontally
260 template<typename DType, typename Storage>
261 std::vector<NDArrayDynamic<DType, Storage>> NDArrayDynamic<DType, Storage>::hsplit(Size index) const {
262 auto sz = size();
263 if (sz == 0)
264 throw std::runtime_error("Cannot hsplit an empty array");
265 if (index > sz) {
266 throw std::runtime_error("Index exceeds array bounds");
267 }
268 Shape sh = shape();
269 std::size_t splitIndex;
270 if (sh.size() == 1) {
271 splitIndex = 0;
272 } else {
273 splitIndex = 1;
274 }
275 Shape sh1{sh};
276 sh1[splitIndex] = index;
277 Shape sh2{sh};
278 auto rest = sh[splitIndex] - index;
279 sh2[splitIndex] = rest;
280 NDArrayDynamic<DType, Storage> result1{sh1};
281 NDArrayDynamic<DType, Storage> result2{sh2};
282 Size i = 0;
283 Size i1 = 0;
284 Size i2 = 0;
285 while (i < size()) {
286 std::copy(m_ArrayImpl.m_Impl.begin() + i,
287 m_ArrayImpl.m_Impl.begin() + i + index,
288 result1.m_ArrayImpl.m_Impl.begin() + i1);
289 i1 += index;
290 i += index;
291 std::copy(m_ArrayImpl.m_Impl.begin() + i,
292 m_ArrayImpl.m_Impl.begin() + i + rest,
293 result2.m_ArrayImpl.m_Impl.begin() + i2);
294 i2 += rest;
295 i += rest;
296 }
297 return {result1, result2};
298 }

300 // Split the array vertically
301 template<typename DType, typename Storage>
302 std::vector<NDArrayDynamic<DType, Storage>> NDArrayDynamic<DType, Storage>::vsplit(Size index) const {
303 auto sz = size();
304 if (sz == 0)
305 throw std::runtime_error("Cannot vsplit an empty array");
306 Shape sh = shape();
307 if (sh.size() < 2) {
308 throw std::runtime_error("vsplit only works on arrays of 2 or more dimensions");
309 }
310 if (index > len() - 1) {
311 throw std::runtime_error("Index exceeds array bounds");
312 }
313 Shape sh1{sh};
314 sh1[0] = index;
315 Shape sh2{sh};
316 auto rest = sh[0] - index;
317 sh2[0] = rest;
318 NDArrayDynamic<DType, Storage> result1{sh1};
319 NDArrayDynamic<DType, Storage> result2{sh2};
320 Size i = 0;
321 std::copy(m_ArrayImpl.m_Impl.begin(),
322 m_ArrayImpl.m_Impl.begin() + sh[1] * index,
323 result1.m_ArrayImpl.m_Impl.begin());
324 i += sh[1] * index;
325 std::copy(m_ArrayImpl.m_Impl.begin() + i,
326 m_ArrayImpl.m_Impl.begin() + i + sh[1] * rest,
327 result2.m_ArrayImpl.m_Impl.begin());
328 return {result1, result2};
329 }
