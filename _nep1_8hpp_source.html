43 inline static void writeNep1Header(std::ostream& stream, const std::string& dType, const std::string& shape) {
44 // The first 6 bytes are a magic string: exactly “x93NUMPY”.
45 static const char* magic = "\x93NUMPY";
46 static const uint32_t magicLen = sizeof(magic) - 1;
47 stream << magic;
48 // The next 1 byte is an unsigned byte: the major version number of the file format, e.g. x01.
49 uint8_t major = 1;
50 // The next 1 byte is an unsigned byte: the minor version number of the file format, e.g. x00.
51 // Note: the version of the file format is not tied to the version of the numpy package.
52 uint8_t minor = 0;
53 stream << major << minor;
54 // “descr” dtype.descr
55 // An object that can be passed as an argument to the numpy.dtype() constructor to create the array’s dtype.
56 // “fortran_order” bool
57 // Whether the array data is Fortran-contiguous or not. Since Fortran-contiguous arrays are a common form of non-C-contiguity, we allow them to be written directly to disk for efficiency.
58 // “shape” tuple of int
59 // The shape of the array.
60 std::string dTypeStr = "{\'descr\': \'" + dType + "\', ";
61
62 dTypeStr += R"('fortran_order': False, 'shape': ()" + shape + "), }";
63 // The next 2 bytes form a little-endian unsigned short int: the length of the header data HEADER_LEN.
64 uint16_t headerLen = 0x80 - magicLen - sizeof(major) - sizeof(minor) - sizeof(uint16_t) + 1; // version 2 format if needed
65 stream.write((char*)&headerLen, sizeof(headerLen));
66 // The next HEADER_LEN bytes form the header data describing the array’s format.
67 // It is an ASCII string which contains a Python literal expression of a dictionary.
68 // It is terminated by a newline (’n’) and padded with spaces (’x20’) to make the total length of the magic
69 // string + 4 + HEADER_LEN be evenly divisible by 16 for alignment purposes.
70 stream << dTypeStr;
71 // padding
72 uint32_t paddingLen = headerLen - dTypeStr.length() - 1;
73 for (std::size_t i = 0; i < paddingLen; ++i) {
74 stream << " ";
75 }
76 stream << std::endl;
77 }

79 inline static std::string parseDescr(const std::string& header) {
80 // “descr” dtype.descr
81 // An object that can be passed as an argument to the numpy.dtype() constructor to create the array’s dtype.
82 // “fortran_order” bool
83 // Whether the array data is Fortran-contiguous or not. Since Fortran-contiguous arrays are a common form of non-C-contiguity, we allow them to be written directly to disk for efficiency.
84 static constexpr char descrPattern[] = "\'descr\': \'";
85 std::size_t descrStart = header.find(descrPattern);
86 if (descrStart == std::string::npos) {
87 throw std::runtime_error("Array DType description is not found");
88 }
89 descrStart += sizeof(descrPattern) - 1;
90 std::size_t descrEnd = header.find('\'', descrStart);
91 if (descrEnd == std::string::npos) {
92 throw std::runtime_error("Array DType description has incorrect format");
93 }
94 std::string descr = header.substr(descrStart, descrEnd - descrStart);
95 return descr;
96 }

98 inline static Shape parseShape(const std::string& header) {
99 // “shape” tuple of int
100 // The shape of the array.
101
102 static constexpr char shapePattern[] = "\'shape\': (";
103 std::size_t shapeStart = header.find(shapePattern);
104 if (shapeStart == std::string::npos) {
105 throw std::runtime_error("Array shape is not found");
106 }
107 shapeStart += sizeof(shapePattern) - 1;
108 std::size_t shapeEnd = header.find(')', shapeStart);
109 if (shapeEnd == std::string::npos) {
110 throw std::runtime_error("Array DType description has incorrect format");
111 }
112 std::string shapeStr{header.substr(shapeStart, shapeEnd - shapeStart)};
113 return Shape{shapeStr};
114 }

116 inline static std::tuple<std::string, Shape> readNep1Header(std::istream& stream) {
117 // The first 6 bytes are a magic string: exactly “x93NUMPY”.
118 static const char magic[] = "\x93NUMPY";
119 static constexpr uint32_t magicLen = std::size(magic) - 1;
120 char magicRead[magicLen];
121 stream.read(magicRead, magicLen);
122 NP_THROW_UNLESS(std::memcmp(magic, magicRead, magicLen) == 0, "Invalid magic");
123 // The next 1 byte is an unsigned byte: the major version number of the file format, e.g. x01.
124 static constexpr uint8_t major = 1;
125 uint8_t majorRead;
126 stream.read((char*)&majorRead, 1);
127 if (majorRead != major) {
128 throw std::runtime_error("Invalid major");
129 }
130 // The next 1 byte is an unsigned byte: the minor version number of the file format, e.g. x00.
131 // Note: the version of the file format is not tied to the version of the numpy package.
132 static constexpr uint8_t minor = 0;
133 uint8_t minorRead;
134 stream.read((char*)&minorRead, 1);
135 if (minorRead != minor) {
136 throw std::runtime_error("Invalid minor");
137 }
138
139 uint16_t headerSize;
140 stream.read((char*)&headerSize, sizeof(headerSize));
141 std::string header;
142 header.resize(headerSize);
143 stream.read(header.data(), headerSize);
144
145 auto descr = parseDescr(header);
146 auto shape = parseShape(header);
147
148 return std::make_tuple(descr, shape);
149 }
