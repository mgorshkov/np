56  inline static Descr parseDescr(const std::string &header) {
57  // “descr” dtype.descr
58  // An object that can be passed as an argument to the numpy.dtype() constructor to create the array’s dtype.
59  // “fortran_order” bool
60  // Whether the array data is Fortran-contiguous or not. Since Fortran-contiguous arrays are a common
61  // form of non-C-contiguity, we allow them to be written directly to disk for efficiency.
62  static constexpr char descrPattern[] = "\'descr\': \'";
63  std::size_t descrStart = header.find(descrPattern);
64  if (descrStart == std::string::npos) {
65  throw std::runtime_error("Array DType description is not found");
66  }
67  descrStart += sizeof(descrPattern) - 1;
68  std::size_t descrEnd = header.find('\'', descrStart);
69  if (descrEnd == std::string::npos) {
70  throw std::runtime_error("Array DType description has incorrect format");
71  }
72  std::string descrStr = header.substr(descrStart, descrEnd - descrStart);
73  Descr descr{};
74  descr.byteOrder = descrStr[0];
75  descr.name = descrStr[1];
76  descr.size = std::stol(descrStr.substr(2, descrStr.size() - 2));
77  return descr;
78  }

168  inline static void
169  writeNep1Header(std::ostream &stream, const Descr &descr, const std::string &shape) {
170  // The first 6 bytes are a magic string: exactly “x93NUMPY”.
171  static const char magic[] = "\x93NUMPY";
172  static const uint32_t magicLen = sizeof(magic) - 1;
173  stream << magic;
174  // The next 1 byte is an unsigned byte: the major version number of the file format, e.g. x01.
175  uint8_t major = 1;
176  // The next 1 byte is an unsigned byte: the minor version number of the file format, e.g. x00.
177  // Note: the version of the file format is not tied to the version of the numpy package.
178  uint8_t minor = 0;
179  stream << major << minor;
180  // “descr” dtype.descr
181  // An object that can be passed as an argument to the numpy.dtype() constructor to create the array’s dtype.
182  // “fortran_order” bool
183  // Whether the array data is Fortran-contiguous or not. Since Fortran-contiguous arrays are a common form of non-C-contiguity, we allow them to be written directly to disk for efficiency.
184  // “shape” tuple of int
185  // The shape of the array.
186  std::ostringstream dType;
187  descr >> dType;
188  std::string dTypeStr = "{" + dType.str() + ", ";
189 
190  dTypeStr += R"('fortran_order': False, 'shape': ()" + shape + "), }";
191  // The next 2 bytes form a little-endian unsigned short int: the length of the header data HEADER_LEN.
192  uint16_t headerLen = 0x80 - magicLen - sizeof(major) - sizeof(minor) - sizeof(uint16_t);// version 2 format if needed
193  stream.write((char *) &headerLen, sizeof(headerLen));
194  // The next HEADER_LEN bytes form the header data describing the array’s format.
195  // It is an ASCII string which contains a Python literal expression of a dictionary.
196  // It is terminated by a newline (’n’) and padded with spaces (’x20’) to make the total length of the magic
197  // string + 4 + HEADER_LEN be evenly divisible by 16 for alignment purposes.
198  stream << dTypeStr;
199  // padding
200  uint32_t paddingLen = headerLen - dTypeStr.length() - 1;
201  for (std::size_t i = 0; i < paddingLen; ++i) {
202  stream << " ";
203  }
204  stream << std::endl;
205  }

207  inline static Shape parseShape(const std::string &header) {
208  // “shape” tuple of int
209  // The shape of the array.
210 
211  static constexpr char shapePattern[] = "\'shape\': (";
212  std::size_t shapeStart = header.find(shapePattern);
213  if (shapeStart == std::string::npos) {
214  throw std::runtime_error("Array shape is not found");
215  }
216  shapeStart += sizeof(shapePattern) - 1;
217  std::size_t shapeEnd = header.find(')', shapeStart);
218  if (shapeEnd == std::string::npos) {
219  throw std::runtime_error("Array DType description has incorrect format");
220  }
221  std::string shapeStr{header.substr(shapeStart, shapeEnd - shapeStart)};
222  return Shape{shapeStr};
223  }

225  inline static std::tuple<Descr, Shape> readNep1Header(std::istream &stream) {
226  // The first 6 bytes are a magic string: exactly “x93NUMPY”.
227  static const char magic[] = "\x93NUMPY";
228  static constexpr uint32_t magicLen = std::size(magic) - 1;
229  char magicRead[magicLen];
230  stream.read(magicRead, magicLen);
231  NP_THROW_UNLESS(std::memcmp(magic, magicRead, magicLen) == 0, "Invalid magic");
232  // The next 1 byte is an unsigned byte: the major version number of the file format, e.g. x01.
233  static constexpr uint8_t major = 1;
234  uint8_t majorRead;
235  stream.read((char *) &majorRead, 1);
236  if (majorRead != major) {
237  throw std::runtime_error("Invalid major");
238  }
239  // The next 1 byte is an unsigned byte: the minor version number of the file format, e.g. x00.
240  // Note: the version of the file format is not tied to the version of the numpy package.
241  static constexpr uint8_t minor = 0;
242  uint8_t minorRead;
243  stream.read((char *) &minorRead, 1);
244  if (minorRead != minor) {
245  throw std::runtime_error("Invalid minor");
246  }
247 
248  uint16_t headerSize;
249  stream.read((char *) &headerSize, sizeof(headerSize));
250  std::string header;
251  header.resize(headerSize);
252  stream.read(header.data(), headerSize);
253 
254  auto descr = parseDescr(header);
255  auto shape = parseShape(header);
256 
257  return std::make_tuple(descr, shape);
258  }
