31 // Array-wise sum
32 template<typename DType, Size SizeT, Size... SizeTs>
33 inline DType NDArrayStatic<DType, SizeT, SizeTs...>::sum() const {
34 DType result{};
35 for (Size i = 0; i < SizeT; ++i) {
36 const auto& element = at(i);
37 result += element.sum();
38 }
39 return result;
40 }
41
42 // Array-wise minimum value
43 template<typename DType, Size SizeT, Size... SizeTs>
44 inline DType NDArrayStatic<DType, SizeT, SizeTs...>::min() const {
45 DType result{};
46 bool inited{false};
47 for (Size i = 0; i < SizeT; ++i) {
48 const auto& element = at(i);
49 DType min_ = element.min();
50 if (!inited || min_ < result) {
51 result = min_;
52 inited = true;
53 }
54 }
55 return result;
56 }

70 }
