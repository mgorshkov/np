212 template<typename DType = DTypeDefault>
213 auto linspace(DType start, DType stop, Size num = 50) {
214 NP_THROW_UNLESS(num > 0, "Number of samples must be non-negative.");
215
216 std::vector<DType> vector;
217 vector.reserve(num);
218 Size i{0};
219 const DType delta = (stop - start) / (num - 1);
220 for (DType value = start; value <= stop; value += delta) {
221 vector.set(i++, value);
222 }
223 return NDArrayDynamic<DType>{vector};
224 }
