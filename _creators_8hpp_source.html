158 template<typename DType = DTypeDefault>
159 auto arange(DType start, DType stop, DType step) {
160 NP_THROW_UNLESS(step != 0, "Step must not be zero.");
161
162 DType const size = (stop - start) / step;
163 std::array<DType, size> array;
164 Size i{0};
165 for (DType value = start; value < stop; value += step) {
166 array[i++] = value;
167 }
168 return NDArrayDynamic<DType>{array};

210 template<typename DType = DTypeDefault>
211 auto linspace(DType start, DType stop, Size num = 50) {
212 NP_THROW_UNLESS(num > 0, "Number of samples must be non-negative.");
213
214 std::vector<DType> vector;
215 vector.reserve(num);
216 Size i{0};
217 const DType delta = (stop - start) / (num - 1);
218 for (DType value = start; value <= stop; value += delta) {
219 vector.set(i++, value);
220 }
221 return NDArrayDynamic<DType>{vector};
222 }

300 auto eye(Size size) {
301 std::vector<std::vector<DType>> vector;
302 vector.resize(size);
303 for (Size i = 0; i < size; ++i) {
304 vector[i].resize(size);
305 for (Size j = 0; j < size; ++j) {
306 vector[i][j] = i == j;
307 }
308 }
309 return NDArrayDynamic<DType>{vector};
