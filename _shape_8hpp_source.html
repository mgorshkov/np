115  explicit Shape(const std::string &shapeTupleStr) {
116  // 0,
117  // 1,
118  // 1, 2
119  // 1, 2, 3
120  if (shapeTupleStr == "0,")
121  return;// empty
122  Size prevCommaIndex = 0;
123  auto push_number = [this, &shapeTupleStr](Size commaIndex, Size prevCommaIndex) {
124  if (commaIndex == prevCommaIndex)
125  return;
126  auto number{shapeTupleStr.substr(prevCommaIndex, commaIndex - prevCommaIndex)};
127  std::size_t i = 0;
128  while (!std::isdigit(number[i]) && i < number.length()) {
129  ++i;
130  }
131  auto numberStr{number.substr(i, number.length() - i)};
132  push_back(std::stoul(numberStr));
133  };
134  while (true) {
135  std::size_t commaIndex = shapeTupleStr.find(',', prevCommaIndex);
136  if (commaIndex == std::string::npos) {
137  push_number(shapeTupleStr.length(), prevCommaIndex);
138  break;
139  }
140  push_number(commaIndex, prevCommaIndex);
141  prevCommaIndex = commaIndex + 1;
142  }
143  NP_THROW_UNLESS_WITH_ARG(!empty(), "Incorrect shape string format", shapeTupleStr);
144  }
