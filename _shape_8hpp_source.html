82 inline explicit Shape(const std::string& shapeTupleStr) {
83 // 0,
84 // 1,
85 // 1, 2
86 // 1, 2, 3
87 if (shapeTupleStr == "0,")
88 return; // empty
89 Size prevCommaIndex = 0;
90 auto push_number = [this, &shapeTupleStr](Size commaIndex, Size prevCommaIndex) {
91 if (commaIndex == prevCommaIndex)
92 return;
93 auto number{shapeTupleStr.substr(prevCommaIndex, commaIndex - prevCommaIndex)};
94 std::size_t i = 0;
95 while (!std::isdigit(number[i]) && i < number.length()) {
96 ++i;
97 }
98 auto numberStr{number.substr(i, number.length() - i)};
99 push_back(std::stoul(numberStr));
100 };
101 while (true) {
102 std::size_t commaIndex = shapeTupleStr.find(',', prevCommaIndex);
103 if (commaIndex == std::string::npos) {
104 push_number(shapeTupleStr.length(), prevCommaIndex);
105 break;
106 }
107 push_number(commaIndex, prevCommaIndex);
108 prevCommaIndex = commaIndex + 1;
109 }
110 NP_THROW_UNLESS_WITH_ARG(!empty(), "Incorrect shape string format", shapeTupleStr);
111 }

121 inline explicit operator std::string() const {
122 if (empty()) {
123 return "0,";
124 } else if (size() == 1) {
125 return std::to_string(at(0)) + ",";

128 std::string shape;
129 for (std::size_t dim = 0; dim < size(); ++dim) {

131 shape += ", ";
132 shape += std::to_string(at(dim));
133 }
